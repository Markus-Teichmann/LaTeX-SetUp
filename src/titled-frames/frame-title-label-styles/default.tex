\renewcommand{\setFrameLabelStyle}[1]{%
    %#1 ist der Name der Umgebung z.b. Definition oder Satz.
    \BeforeBeginEnvironment{#1}{%
        \IfEndWith{\thesubsection}{0}
        {
            \refstepcounter{#1}
            \phantomsection
            \hypertarget{#1:\thesection.\arabic{#1}}
        }
        %else:
        {
            \refstepcounter{#1}
            \phantomsection
            \hypertarget{#1:\thesubsection.\arabic{#1}}
        }
    }
}
