\pgfkeys{%
    /tbl/.unknown/.code = { \relax },
    /tbl/.cd,
        caption/.code = {%
            \ifx#1\empty\else
                \def\tblCaption{#1}
            \fi
        },
        citation/.code = {%
            \ifx#1\empty\else
                \def\tblCitation{#1}
            \fi
        }
}

\newcounter{tbl}
\setcounter{tbl}{0}
\AtBeginEnvironment{tbl}{
    \stepcounter{tbl}
    %\addtocounter{footnote}{-3}
}

\NewColumnType{C}[1]{%
    Q[
        c,

    ]
}

\NewColumnType{B}[2]{%
    Q[
        halign=#1,
        valign=#2,
        cmd={%
            \bfseries
        }
    ]
}
